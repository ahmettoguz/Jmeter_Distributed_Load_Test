
Başlatma
    kubectl apply -f jmeter-deployment.yaml
    kubectl apply -f jmeter-service.yaml
    
Pod içerisine girme: 
    kubectl exec -it jmeter-566b8976c-tsthq -- /bin/sh
    kubectl exec -it jmeter-master-74795456cd-qdbmf -n jmeter -- /bin/bash
                                                                    
Silme 
    kubectl delete pod jmeter-566b8976c-tsthq -n kube-system

    kubectl delete deployment --all -n jmeter 
    kubectl delete services --all -n jmeter 


Namespace
    kubectl apply -f namespace.yaml
    kubectl get namespaces

Copy:
    kubectl cp ./JmeterDemo_Test_Script.jmx jmeter-master-74795456cd-qdbmf:/ -n jmeter

Servis ve podlar:
    kubectl get pods 
    kubectl get services
    
    kubectl get pods -n jmeter
    kubectl get services -n jmeter
